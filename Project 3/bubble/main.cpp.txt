
#include<iostream>
#include<vector>
#include<list>
#include<algorithm>
#include<cassert>
#include<ctime>
#include "bubble.hpp"

using namespace std;



int main()
{
     vector<int> v;
     //vector<string> v1 = {"abc","cba","ddd","ab","b","a"};
    
     srand(time(NULL));

     for (int i = 0; i < 10000; ++i)
     {
          int num = rand() % 10000 + 1;
          v.push_back(num);
     }

     vector<int> v_test = v;
     clock_t start;
     clock_t end;
     start = clock();
     bubblesort(v.begin(),v.end());
     end = clock();
     cout <<"Bubble sort time = " << end-start <<'\n';
     start = clock();
     std::sort(v_test.begin(),v_test.end());
     end = clock();
     cout <<"std sort time = " << end-start <<'\n';
     assert(v_test == v);
}
